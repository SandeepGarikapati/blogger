## This is arm.jsp and file name should be arm.jsp

```js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Armstrong Number Checker Result</title>
    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            margin: 0;
            background-color: #f4f4f4;
            font-family: Arial, sans-serif;
        }
        h4, p {
            text-align: center;
        }
        .modal {
            display: none; /* Hidden by default */
            position: fixed;
            z-index: 1;
            left: 0;
            top: 0;
            width: 100%;
            height: 100%;
            overflow: auto;
            background-color: rgb(0, 0, 0); /* Fallback color */
            background-color: rgba(0, 0, 0, 0.4); /* Black w/ opacity */
        }
        .modal-content {
            background-color: #fff;
            margin: 15% auto;
            padding: 20px;
            border: 1px solid #888;
            width: 80%;
            max-width: 400px;
            border-radius: 10px;
            box-shadow: 0 4px 8px rgba(0, 0, 0, 0.2);
        }
        .close {
            color: #aaa;
            float: right;
            font-size: 28px;
            font-weight: bold;
        }
        .close:hover,
        .close:focus {
            color: black;
            text-decoration: none;
            cursor: pointer;
        }
        .armstrong {
            color: green;
            font-weight: bold;
        }
        .not-armstrong {
            color: red;
            font-weight: bold;
        }
    </style>
</head>
<body>

<div id="resultModal" class="modal">
    <div class="modal-content">
        <span class="close">&times;</span>
        <h4>Result:</h4>
        <%
        int number = Integer.parseInt(request.getParameter("number"));
        int originalNumber = number;
        int numberOfDigits = String.valueOf(number).length();
        int sum = 0;
        while (number > 0) {
            int digit = number % 10;
            sum += Math.pow(digit, numberOfDigits);
            number /= 10;
        }
        if (sum == originalNumber) {
        %>
        <p class="armstrong">The given number <%= originalNumber %> is an Armstrong number.</p>
        <%
        } else {
        %>
        <p class="not-armstrong">The given number <%= originalNumber %> is not an Armstrong number.</p>
        <%
        }
        %>
    </div>
</div>

<script>
    // Get the modal
    var modal = document.getElementById("resultModal");

    // Get the <span> element that closes the modal
    var span = document.getElementsByClassName("close")[0];

    // When the page loads, open the modal
    window.onload = function() {
        modal.style.display = "block";
    }

    // When the user clicks on <span> (x), close the modal
    span.onclick = function() {
        modal.style.display = "none";
        window.location.href = "index.html"; // Redirect to the index page after closing
    }

    // When the user clicks anywhere outside of the modal, close it
    window.onclick = function(event) {
        if (event.target == modal) {
            modal.style.display = "none";
            window.location.href = "index.html"; // Redirect to the index page after closing
        }
    }
</script>

</body>
</html>

```
