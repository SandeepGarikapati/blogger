## Result.html

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Concatenated Result</title>
</head>
<body>
  <h1>Concatenated Strings</h1>
  <p id="result"></p>

  <script>
    // Function to get query parameters
    function getQueryParam(param) {
      const urlParams = new URLSearchParams(window.location.search);
      return urlParams.get(param);
    }

    // Get the two strings from the URL query parameters
    const string1 = getQueryParam('string1');
    const string2 = getQueryParam('string2');

    // Concatenate the strings and display the result
    if (string1 && string2) {
      const concatenated = string1 + " " + string2;
      document.getElementById("result").textContent = "Concatenated result: " + concatenated;
    } else {
      document.getElementById("result").textContent = "Please enter valid strings.";
    }
  </script>
</body>
</html>
```

```html
    let editIndex = -1; // Index of the row being edited

    function addOrUpdateBook() {
      const title = document.getElementById("title").value;
      const author = document.getElementById("author").value;
      const isbn = document.getElementById("isbn").value;

      const tableBody = document.getElementById("booksTable").querySelector("tbody");

      if (editIndex === -1) {
        // Add new record
        const newRow = tableBody.insertRow();
        newRow.innerHTML = `
          <td>${title}</td>
          <td>${author}</td>
          <td>${isbn}</td>
          <td class="update-row">
            <button onclick="editBook(this)">Update</button>
            <button class="delete" onclick="deleteBook(this)">Delete</button>
          </td>
        `;
      } else {
        // Update existing record
        const row = tableBody.rows[editIndex];
        row.cells[0].textContent = title;
        row.cells[1].textContent = author;
        row.cells[2].textContent = isbn;
        editIndex = -1;
      }

      // Reset form
      document.getElementById("bookForm").reset();
    }

    function editBook(button) {
      const row = button.closest("tr");
      editIndex = row.rowIndex - 1; // Adjust for the header row

      // Populate form fields
      document.getElementById("title").value = row.cells[0].textContent;
      document.getElementById("author").value = row.cells[1].textContent;
      document.getElementById("isbn").value = row.cells[2].textContent;
    }

    function deleteBook(button) {
      const row = button.closest("tr");
      row.remove();
    }
  </script>
</body>
</html>
```
